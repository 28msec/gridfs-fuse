SET(SHELLEXE "bash")
SET(DEPENDENT_LIB_DIRS "${MongoDB_LIBRARY_DIRS}") 

IF (APPLE)
  SET(TIMESTAMP_CMD "stat -f %c-%m-%a")
ELSE(APPLE)
  SET(TIMESTAMP_CMD "stat -c %Z-%Y-%X")
ENDIF(APPLE)

CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/simple.sh.in ${CMAKE_CURRENT_BINARY_DIR}/simple.sh)

SET (TestName "gridfs-fuse-simple")
ADD_TEST(${TestName} ${SHELLEXE} "${CMAKE_CURRENT_BINARY_DIR}/simple.sh" )
IF (APPLE)
  SET_TESTS_PROPERTIES(${TestName}
                       PROPERTIES ENVIRONMENT "DYLD_LIBRARY_PATH=${DEPENDENT_LIB_DIRS}")
ELSE(APPLE)
  SET_TESTS_PROPERTIES(${TestName}
                       PROPERTIES ENVIRONMENT "LD_LIBRARY_PATH=${DEPENDENT_LIB_DIRS}")
ENDIF(APPLE)
